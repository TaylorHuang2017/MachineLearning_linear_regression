def matxMultiply(A, B):
    if len(A[0])!= len(B):
        raise ValueError
        return
    p = [[0] * len(B[0]) for i in range(len(A))]
    for i in range(len(A)):
        for j in range(len(B[0])):
            for k in range(len(B)):
                p[i][j] += A[i][k] * B[k][j]            
    return p

def augmentMatrix(A, b):
    if len(A) != len(b):
        raise ValueError
        return
    
    # result    
    result = []
    for i in range(len(A)):
        # row 
        row = []

            row.append(A[i][j])
        for k in range(len(b[i])):

def gj_Solve(A, b, decPts=4, epsilon=1.0e-16):
    if len(A) != len(b):  # A b 
        raise ValueError   # ValueError
        return
    
    Ab = augmentMatrix(A,b)  #  Ab
    
    for c in range(len(Ab[0])-1):  # 
        Ab_t = transpose(Ab)   # c
        c_list = Ab_t[c]     # 
        max_val = max(c_list[c:],key=abs) # max_val
        if abs(max_val) < epsilon:    # 0 ANone
            return None
        max_row = c + c_list[c:].index(max_val)  # 
        swapRows(Ab,c,max_row)   #  c
        scaleRow(Ab,c,1.0/Ab[c][c])   # c1
        for r in range(len(A)):           # 
            if r != c and Ab[r][c] != 0:    #  c  & 
                addScaledRow(Ab,r,c,-Ab[r][c])  # c0
    matxRound(Ab)    # 
    
    return [[val] for val in transpose(Ab)[-1]]  # Ab

def linearRegression2D(X,Y):
    X = [[v, 1] for v in X]
    Y = [[v] for v in Y]
    X_t = transpose(X)
    A = matxMultiply(X_t, X)
    b = matxMultiply(X_t, Y)
    result = gj_Solve(A, b)    
    return tuple([v[0] for v in result])
